```js filename="Modal.js"  {4-6}
let modal = document.getElementById("modal");
let modalButton = document.getElementById("modal-btn");
let isOpen = false;

modalButton.addEventListener('click', () => {
    isOpen = !isOpen;

    if (isOpen) {
        openModal();
    } else {
        closeModal();
    }
})
```
